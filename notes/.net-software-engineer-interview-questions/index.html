Language Fundamentals

Interface vs Abstract class

// Interface - contract only
public interface IPaymentProcessor
{
    void ProcessPayment(decimal amount);
}

// Abstract class - shared implementation
public abstract class PaymentProcessor
{
    public abstract void ProcessPayment(decimal amount);
    public void LogTransaction() { /* shared logic */ }
}

Value vs Reference types

// Reference type (class) - nullable by default
public class Person { public string Name { get; set; } }

// Value type (struct) - not nullable unless explicit
public struct Point { public int X; public int Y; }
public struct Point? nullablePoint = null; // Nullable value type

async/await in Web API

// Frees thread pool threads during I/O operations
public async Task<ActionResult<User>> GetUser(int id)
{
    var user = await _repository.GetUserAsync(id); // Thread released during DB call
    return Ok(user);
}

const vs readonly

public const int MaxRetries = 3;        // Compile-time constant
public readonly DateTime CreatedAt;     // Runtime constant, set in constructor

yield

public IEnumerable<int> GetNumbers()
{
    yield return 1;
    yield return 2;
    yield return 3; // Lazy evaluation, generates values on demand
}

throw vs throw ex

try { /* code */ }
catch (Exception ex)
{
    throw;    // Preserves stack trace
    throw ex; // Resets stack trace - avoid this
}

IQueryable vs IEnumerable

IQueryable<User> query = dbContext.Users
    .Where(u => u.Age > 18); // Translates to SQL, executes on DB

IEnumerable<User> users = dbContext.Users.AsEnumerable()
    .Where(u => u.Age > 18); // Loads all, filters in memory

Patterns

SOLID Principles

// Single Responsibility
public class OrderService { /* handles orders only */ }

// Dependency Inversion
public class OrderController
{
    private readonly IOrderService _orderService;
    public OrderController(IOrderService orderService) 
        => _orderService = orderService; // Depends on abstraction
}

Builder Pattern

public class QueryBuilder
{
    private string _select;
    private string _from;
    
    public QueryBuilder Select(string fields) { _select = fields; return this; }
    public QueryBuilder From(string table) { _from = table; return this; }
    public string Build() => $"SELECT {_select} FROM {_from}";
}

var query = new QueryBuilder().Select("*").From("Users").Build();

Factory Pattern

public interface IPaymentGateway { }
public class PaymentFactory
{
    public IPaymentGateway Create(string type) => type switch
    {
        "stripe" => new StripeGateway(),
        "paypal" => new PayPalGateway(),
        _ => throw new ArgumentException()
    };
}

DDD

Entity vs Value Object vs Aggregate

// Entity - has identity
public class Order 
{
    public Guid Id { get; private set; }
    public OrderStatus Status { get; private set; }
}

// Value Object - no identity, immutable
public record Money(decimal Amount, string Currency);

// Aggregate - consistency boundary
public class Order // Aggregate Root
{
    private List<OrderLine> _lines = new(); // Only Order can modify lines
    public void AddLine(Product product, int quantity) { /* business rules */ }
}

Common Antipatterns

  • Anemic Domain Model: Entities with only getters/setters, logic in services

Bounded Context

Logical boundary where a domain model applies. Customer in Sales context differs from Customer in Support context.

CQRS

// Command - writes
public record CreateOrderCommand(Guid CustomerId, List<OrderLine> Lines);
public class CreateOrderHandler : IRequestHandler<CreateOrderCommand, Guid> { }

// Query - reads
public record GetOrderQuery(Guid OrderId);
public class GetOrderHandler : IRequestHandler<GetOrderQuery, OrderDto> { }

Use with MediatR for separation of reads/writes, different models, and scalability.

Microservices vs Monolith

Microservices:

  • ✅ Independent deployment, technology freedom, scalability
  • ❌ Complexity, distributed transactions, network latency

Monolith:

  • ✅ Simpler deployment, easier debugging, single database
  • ❌ Tight coupling, scaling challenges

Event Driven Architecture

// Event - something that happened
public record OrderCreatedEvent(Guid OrderId, DateTime CreatedAt);

// Command - instruction to do something
public record CreateOrderCommand(Guid CustomerId);

Messages are generic; Events are past-tense facts; Commands are imperative instructions.

React

Functional vs Class Components

// Functional (preferred) - simpler, hooks support
const UserProfile = ({ userId }) => {
  const [user, setUser] = useState(null);
  useEffect(() => { fetchUser(userId).then(setUser); }, [userId]);
  return <div>{user?.name}</div>;
};

// Class - legacy
class UserProfile extends React.Component { }

Hooks

useState()    // State management
useEffect()   // Side effects
useContext()  // Context consumption
useMemo()     // Memoization
useCallback() // Callback memoization
